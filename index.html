using UnityEngine; using System.Collections; public class AutoShoot : MonoBehaviour { public GameObject projetilPrefab; // O objeto do tiro public Transform pontoDeDisparo; // Onde o tiro nasce public float cadenciaDeTiro = 0.5f; // Segundos entre cada disparo private bool atirando = false; void Update() { // Inicia a rotina de disparo se ainda não estiver atirando if (!atirando) { StartCoroutine(DispararAutomatico()); } } IEnumerator DispararAutomatico() { atirando = true; while (true) // Loop infinito enquanto a nave existir { Instantiate(projetilPrefab, pontoDeDisparo.position, pontoDeDisparo.rotation); // Espera o tempo definido antes do próximo tiro yield return new WaitForSeconds(cadenciaDeTiro); } } }
